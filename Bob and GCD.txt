#include <stdio.h>

int main() {
    int T;
    scanf("%d", &T);

    while (T--) {
        long long K, N;
        scanf("%lld %lld", &K, &N);

        long long total_operations = 0;

        for (long long i = 0; i < N; i++) {
            long long A;
            scanf("%lld", &A);

            long long remainder = A % K;

            if (remainder != 0) {
                long long decrease = remainder;
                long long increase = K - remainder;

                // If decreasing makes value < 1, we must increase
                if (A - decrease < 1)
                    total_operations += increase;
                else
                    total_operations += (decrease < increase ? decrease : increase);
            }
        }

        printf("%lld\n", total_operations);
    }

    return 0;
}
