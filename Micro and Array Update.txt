#include <stdio.h>

int main() {
    int T;
    scanf("%d", &T);

    while (T--) {
        int N;
        long long K;
        scanf("%d %lld", &N, &K);

        long long min_val = 1000000000000000000LL;
        long long x;

        for (int i = 0; i < N; i++) {
            scanf("%lld", &x);
            if (x < min_val)
                min_val = x;
        }

        if (min_val >= K)
            printf("0\n");
        else
            printf("%lld\n", K - min_val);
    }

    return 0;
}
