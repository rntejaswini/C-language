#include <stdio.h>
#include <stdlib.h>

#define MAX 1005

int N, M, Q;
char grid[MAX][MAX];
int dist[MAX][MAX];

int dx[4] = {-1, 1, 0, 0};
int dy[4] = {0, 0, -1, 1};

typedef struct {
    int x, y;
} Node;

Node queue[MAX * MAX];

int main() {
    
    scanf("%d %d %d", &N, &M, &Q);
    
    for(int i = 0; i < N; i++) {
        scanf("%s", grid[i]);
    }
    
    int sx, sy;
    scanf("%d %d", &sx, &sy);
    
    // Convert to 0-based index
    sx--; 
    sy--;
    
    // Initialize distance array
    for(int i = 0; i < N; i++) {
        for(int j = 0; j < M; j++) {
            dist[i][j] = -1;
        }
    }
    
    // BFS
    int front = 0, rear = 0;
    
    if(grid[sx][sy] == 'O') {
        queue[rear++] = (Node){sx, sy};
        dist[sx][sy] = 0;
    }
    
    while(front < rear) {
        Node curr = queue[front++];
        
        for(int i = 0; i < 4; i++) {
            int nx = curr.x + dx[i];
            int ny = curr.y + dy[i];
            
            if(nx >= 0 && nx < N && ny >= 0 && ny < M) {
                if(grid[nx][ny] == 'O' && dist[nx][ny] == -1) {
                    dist[nx][ny] = dist[curr.x][curr.y] + 1;
                    queue[rear++] = (Node){nx, ny};
                }
            }
        }
    }
    
    // Process queries
    while(Q--) {
        int x, y;
        scanf("%d %d", &x, &y);
        
        x--;
        y--;
        
        printf("%d\n", dist[x][y]);
    }
    
    return 0;
}
