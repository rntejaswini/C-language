#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, m;
    scanf("%d", &n);
    int A[n];
    for(int i=0; i<n; i++) scanf("%d", &A[i]);
    
    scanf("%d", &m);
    int C[m];
    for(int i=0; i<m; i++) scanf("%d", &C[i]);
    
    int B[m];  // maximum possible size
    int bSize = 0;
    
    for(int i=0; i<m; i++) {
        int candidate = C[i] - A[0];
        if(candidate <= 0) continue; // only positive integers
        int ok = 1;
        for(int j=0; j<n; j++) {
            int found = 0;
            for(int k=0; k<m; k++) {
                if(A[j] + candidate == C[k]) {
                    found = 1;
                    break;
                }
            }
            if(!found) { ok = 0; break; }
        }
        if(ok) B[bSize++] = candidate;
    }
    
    // Print B in increasing order
    // Simple bubble sort for small arrays
    for(int i=0; i<bSize-1; i++) {
        for(int j=i+1; j<bSize; j++) {
            if(B[i] > B[j]) {
                int temp = B[i]; B[i] = B[j]; B[j] = temp;
            }
        }
    }
    
    for(int i=0; i<bSize; i++) {
        if(i>0) printf(" ");
        printf("%d", B[i]);
    }
    printf("\n");
    
    return 0;
}
