#include <stdio.h>
#include <stdlib.h>

int compare(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

int main() {
    int T;
    scanf("%d", &T);
    
    while(T--) {
        int N, M;
        scanf("%d %d", &N, &M);
        
        int arr[1000];
        int totalSum = 0;
        
        for(int i = 0; i < N; i++) {
            scanf("%d", &arr[i]);
            totalSum += arr[i];
        }
        
        // Sort array
        qsort(arr, N, sizeof(int), compare);
        
        int sumSmallest = 0;
        int sumLargest = 0;
        
        // Sum of M smallest elements
        for(int i = 0; i < M; i++) {
            sumSmallest += arr[i];
        }
        
        // Sum of M largest elements
        for(int i = N - M; i < N; i++) {
            sumLargest += arr[i];
        }
        
        int maxSum = totalSum - sumSmallest;
        int minSum = totalSum - sumLargest;
        
        printf("%d\n", maxSum - minSum);
    }
    
    return 0;
}
