#include<stdio.h>
#include<stdlib.h>

int cmpfunc(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

int Max(int N, int K, int* arr) {
    
    int odd = 0, even = 0;
    int *costs = (int*)malloc(sizeof(int) * N);
    int count = 0;
    
    // Find valid cut positions
    for(int i = 0; i < N - 1; i++) {
        
        if(arr[i] % 2 == 0)
            even++;
        else
            odd++;
        
        if(odd == even) {
            costs[count++] = abs(arr[i] - arr[i+1]);
        }
    }
    
    // Sort costs
    qsort(costs, count, sizeof(int), cmpfunc);
    
    int total = 0;
    int cuts = 0;
    
    // Pick minimum cost cuts first
    for(int i = 0; i < count; i++) {
        if(total + costs[i] <= K) {
            total += costs[i];
            cuts++;
        } else {
            break;
        }
    }
    
    free(costs);
    
    return cuts;
}

int main() {
    int N;
    scanf("%d", &N);
    
    int K;
    scanf("%d", &K);
    
    int *arr = (int *)malloc(sizeof(int) * N);
    
    for(int i = 0; i < N; i++)
        scanf("%d", &arr[i]);

    int out_ = Max(N, K, arr);
    printf("%d", out_);
    
    free(arr);
    return 0;
}
