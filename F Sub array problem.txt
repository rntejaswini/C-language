#include <stdio.h>
#include <math.h>

#define MAXN 1000005

// Function to check prime
int isPrime(int x) {
    if (x < 2) return 0;
    if (x == 2) return 1;
    if (x % 2 == 0) return 0;
    for (int i = 3; i * i <= x; i += 2) {
        if (x % i == 0)
            return 0;
    }
    return 1;
}

int main() {
    long long N, K;
    scanf("%lld %lld", &N, &K);

    long long arr[N];

    for (int i = 0; i < N; i++) {
        long long x;
        scanf("%lld", &x);
        arr[i] = isPrime(x);  // convert to 0 or 1
    }

    long long left = 0;
    long long sum = 0;
    long long answer = 0;

    for (long long right = 0; right < N; right++) {

        sum += arr[right];

        while (sum > K) {
            sum -= arr[left];
            left++;
        }

        answer += (right - left + 1);
    }

    printf("%lld\n", answer);

    return 0;
}
