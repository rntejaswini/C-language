#include <stdio.h>

int main() {
    int T;
    scanf("%d", &T);

    while(T--) {

        int N;
        long long X, Y;
        scanf("%d %lld %lld", &N, &X, &Y);

        int arr[N];

        for(int i = 0; i < N; i++)
            scanf("%d", &arr[i]);

        long long cost1 = 0;
        long long cost2 = 0;

        for(int i = 0; i < N; i++) {

            int expected1 = i % 2;        // pattern 0 1 0 1
            int expected2 = (i + 1) % 2;  // pattern 1 0 1 0

            if(arr[i] != expected1) {
                if(arr[i] == 0)
                    cost1 += X;
                else
                    cost1 += Y;
            }

            if(arr[i] != expected2) {
                if(arr[i] == 0)
                    cost2 += X;
                else
                    cost2 += Y;
            }
        }

        printf("%lld\n", cost1 < cost2 ? cost1 : cost2);
    }

    return 0;
}
