#include <stdio.h>
#include <string.h>
#include <math.h>

#define MAXN 20

char s[MAXN];
int n;
int dp[MAXN][MAXN];
int primeMemo[MAXN][MAXN]; // -1 unknown, 0 not prime, 1 prime

long long getNumber(int l, int r) {
    long long num = 0;
    for (int i = l; i <= r; i++) {
        num = num * 10 + (s[i] - '0');
    }
    return num;
}

int isPrimeLL(long long x) {
    if (x < 2) return 0;
    for (long long i = 2; i * i <= x; i++) {
        if (x % i == 0) return 0;
    }
    return 1;
}

int isPrimeSub(int l, int r) {
    if (primeMemo[l][r] != -1)
        return primeMemo[l][r];

    long long val = getNumber(l, r);
    primeMemo[l][r] = isPrimeLL(val);
    return primeMemo[l][r];
}

int main() {
    int T;
    scanf("%d", &T);

    while (T--) {
        scanf("%s", s);
        n = strlen(s);

        memset(dp, 0, sizeof(dp));
        memset(primeMemo, -1, sizeof(primeMemo));

        // length 1 substrings → losing
        for (int i = 0; i < n; i++)
            dp[i][i] = 0;

        // increasing length
        for (int len = 2; len <= n; len++) {
            for (int l = 0; l + len - 1 < n; l++) {
                int r = l + len - 1;
                dp[l][r] = 0;

                // remove front
                if (!isPrimeSub(l + 1, r) && dp[l + 1][r] == 0)
                    dp[l][r] = 1;

                // remove back
                if (!dp[l][r] && !isPrimeSub(l, r - 1) && dp[l][r - 1] == 0)
                    dp[l][r] = 1;
            }
        }

        if (dp[0][n - 1])
            printf("Alice\n");
        else
            printf("Bob\n");
    }

    return 0;
}
