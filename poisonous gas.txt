#include <stdio.h>

int isPowerOfTwo(int x) {
    return x > 0 && (x & (x - 1)) == 0;
}

int main() {
    int T;
    scanf("%d", &T);
    
    while (T--) {
        int N;
        scanf("%d", &N);
        
        int A[N];
        int sum = 0;
        for (int i = 0; i < N; i++) {
            scanf("%d", &A[i]);
            if (A[i] > 0) sum += A[i];
        }
        
        if (isPowerOfTwo(sum)) {
            printf("Yes\n");
        } else {
            printf("No\n");
        }
    }
    
    return 0;
}
