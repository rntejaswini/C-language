#include <stdio.h>

int countSetBits(long long n) {
    int count = 0;
    while (n) {
        count += (n & 1);
        n >>= 1;
    }
    return count;
}

int main() {
    int T;
    scanf("%d", &T);

    while (T--) {
        long long N;
        scanf("%lld", &N);

        long long best = N;
        int maxBits = countSetBits(N);

        long long val = 1;

        while (val <= N) {
            int bits = countSetBits(val);
            if (bits > maxBits) {
                maxBits = bits;
                best = val;
            }
            val = (val << 1) | 1;   // Generate 1, 3, 7, 15, 31...
        }

        printf("%lld\n", best);
    }

    return 0;
}
