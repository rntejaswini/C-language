#include <stdio.h>
#include <stdlib.h>

int compare(const void *a, const void *b) {
    long long x = *(long long*)a;
    long long y = *(long long*)b;
    if (x < y) return -1;
    if (x > y) return 1;
    return 0;
}

int main() {
    int N;
    scanf("%d", &N);

    long long A[N];

    for (int i = 0; i < N; i++)
        scanf("%lld", &A[i]);

    // Sort the array
    qsort(A, N, sizeof(long long), compare);

    // Prefix sums
    long long prefix[N + 1];
    prefix[0] = 0;

    for (int i = 1; i <= N; i++)
        prefix[i] = prefix[i - 1] + A[i - 1];

    int Q;
    scanf("%d", &Q);

    while (Q--) {
        long long K;
        scanf("%lld", &K);

        int left = 1, right = N;
        int ans = 0;

        while (left <= right) {
            int mid = (left + right) / 2;

            if (prefix[mid] < K * mid) {
                ans = mid;
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }

        printf("%d\n", ans);
    }

    return 0;
}
