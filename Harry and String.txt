#include <stdio.h>
#include <string.h>
#include <stdbool.h>

bool isVowel(char c) {
    return (c=='a'||c=='e'||c=='i'||c=='o'||c=='u');
}

int main() {
    int n;
    scanf("%d", &n);
    getchar(); // consume newline

    char word[1000001]; // max word length ~1e6
    for(int t=0; t<n; t++) {
        fgets(word, sizeof(word), stdin);

        // Remove newline if present
        int len = strlen(word);
        if(word[len-1] == '\n') word[len-1] = '\0';

        char vowels[1000001];
        int vcount = 0;

        // Extract vowels
        for(int i=0; word[i]; i++) {
            if(isVowel(word[i])) {
                vowels[vcount++] = word[i];
            }
        }

        if(vcount == 0) {
            printf("Good\n"); // No vowels → Good
            continue;
        }

        // Check ascending
        bool ascending = true, descending = true;
        for(int i=1; i<vcount; i++) {
            if(vowels[i] < vowels[i-1]) ascending = false;
            if(vowels[i] > vowels[i-1]) descending = false;
        }

        if(ascending) printf("Good\n");
        else if(descending) printf("Worst\n");
        else printf("Bad\n");
    }

    return 0;
}
