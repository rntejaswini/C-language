var cloneGraph = function(node) {
    if (!node) return null;
    
    const map = new Map(); // original -> clone
    
    function dfs(curr) {
        // If already cloned, return it
        if (map.has(curr)) {
            return map.get(curr);
        }
        
        // Create clone
        const clone = new _Node(curr.val);
        map.set(curr, clone);
        
        // Clone neighbors
        for (let neighbor of curr.neighbors) {
            clone.neighbors.push(dfs(neighbor));
        }
        
        return clone;
    }
    
    return dfs(node);
};
