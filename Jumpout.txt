#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    int A[10005];

    for(int i = 1; i <= N; i++) {
        scanf("%d", &A[i]);
    }

    for(int i = 1; i <= N; i++) {
        if(i + A[i] > N) {
            printf("%d\n", i);
            break;
        }
    }

    return 0;
}
