#include <stdio.h>
#include <stdlib.h>

int cmp_asc(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

int cmp_desc(const void *a, const void *b) {
    return (*(int*)b - *(int*)a);
}

int main() {

    int n;
    scanf("%d", &n);

    int bob[n], james[n];

    for(int i = 0; i < n; i++)
        scanf("%d", &bob[i]);

    for(int i = 0; i < n; i++)
        scanf("%d", &james[i]);

    qsort(bob, n, sizeof(int), cmp_desc);
    qsort(james, n, sizeof(int), cmp_asc);

    long long score = 0;

    for(int i = 0; i < n; i++) {
        if(bob[i] > james[i])
            score += bob[i] - james[i];
    }

    printf("%lld\n", score);

    return 0;
}
