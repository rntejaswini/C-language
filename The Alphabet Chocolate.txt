#include <stdio.h>
#include <string.h>

int isVowel(char c) {
    return (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u' ||
            c == 'A' || c == 'E' || c == 'I' || c == 'O' || c == 'U');
}

int main() {
    int T;
    scanf("%d", &T);

    while (T--) {
        char str[100005];
        scanf("%s", str);

        long long total = 0;
        int n = strlen(str);

        for (int i = 0; i < n; i++) {
            if (isVowel(str[i])) {
                total += (long long)(i + 1) * (n - i);
            }
        }

        printf("%lld\n", total);
    }

    return 0;
}
