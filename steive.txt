#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int A[n], B[n];
    for(int i = 0; i < n; i++)
        scanf("%d", &A[i]);
    for(int i = 0; i < n; i++)
        scanf("%d", &B[i]);

    int result[n];

    for(int i = 0; i < n; i++) {
        int maxB = B[i];
        for(int j = 0; j < n; j++) {
            if(i != j && (A[i] == A[j] || (i + j + 2 == n + 1))) {
                // i+j+2 because arrays are 0-indexed but the problem assumes 1-indexed
                if(B[j] > maxB)
                    maxB = B[j];
            }
        }
        result[i] = maxB;
    }

    for(int i = 0; i < n; i++) {
        printf("%d ", result[i]);
    }
    printf("\n");

    return 0;
}
